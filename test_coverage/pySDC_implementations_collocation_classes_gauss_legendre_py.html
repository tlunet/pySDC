48 """ 

49 Computes nodes for the Gauss-Legendre quadrature 

50 

51 Python version by Dieter Moser, 2014 

52 

53 Returns: 

54 np.ndarray: array of Gauss-Legendre nodes

56 M = self.num_nodes 

57 a = self.tleft 

58 b = self.tright 

59 

60 # Building the companion matrix comp_mat with det(nodes*I-comp_mat)=P_n(nodes), where P_n is the 

61 # Legendre polynomial under consideration. comp_mat will be constructed in such a way that it is symmetric. 

62 linspace = np.linspace(1, M - 1, M - 1) 

63 v = [linspace[i] / np.sqrt(4.0 * linspace[i] ** 2 - 1.0) for i in range(len(linspace))] 

64 comp_mat = np.diag(v, 1) + np.diag(v, -1) 

65 

66 # Determining the abscissas (nodes) - since det(nodesI-comp_mat)=P_n(nodes), the abscissas are the roots 

67 # of the characteristic polynomial, i.e. the eigenvalues of comp_mat 

68 [eig_vals, _] = np.linalg.eig(comp_mat) 

69 indizes = np.argsort(eig_vals) 

70 nodes = eig_vals[indizes] 

71 

72 # take real part and shift from [-1,1] to [a,b] 

73 nodes = nodes.real 

74 nodes = (a * (1 - nodes) + b * (1 + nodes)) / 2 

75 

76 return nodes
