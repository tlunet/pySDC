52def run_imex_sdc(S): 

53 """ 

54 Routine to run IMEX SDC on a single time step 

55 

56 Args: 

57 S: an instance of a step representing the time step 

58 

59 Returns: 

60 the error of SDC vs. exact solution 

61 the residual after the SDC sweeps 

62 the number of iterations 

63 """ 

64 # make shortcuts for the level and the problem 

65 L = S.levels[0] 

66 P = L.prob 

67 

68 # set initial time in the status of the level 

69 L.status.time = 0.1 

70 # compute initial value (using the exact function here) 

71 L.u[0] = P.u_exact(L.time) 

72 

73 # access the sweeper's predict routine to get things started 

74 # if we don't do this, the values at the nodes are not initialized 

75 L.sweep.predict() 

76 # compute the residual (we may be done already!) 

77 L.sweep.compute_residual() 

78 

79 # reset iteration counter 

80 S.status.iter = 0 

81 # run the SDC iteration until either the maximum number of iterations is reached or the residual is small enough 

82 f = open('step_2_B_out.txt', 'w') 

83 while S.status.iter < S.params.maxiter and L.status.residual > L.params.restol: 

84 # this is where the nodes are actually updated according to the SDC formulas 

85 L.sweep.update_nodes() 

86 # compute/update the residual 

87 L.sweep.compute_residual() 

88 # increment the iteration counter 

89 S.status.iter += 1 

90 out = 'Time %4.2f of %s -- Iteration: %2i -- Residual: %12.8e' \ 

91 % (L.time, L.level_index, S.status.iter, L.status.residual) 

92 f.write(out + '\n') 

93 print(out) 

94 f.close() 

95 

96 # compute the interval's endpoint: this (and only this) will set uend, depending on the collocation nodes 

97 L.sweep.compute_end_point() 

98 # update the simulation time 

99 L.status.time += L.dt 

100 

101 # compute exact solution and compare 

102 uex = P.u_exact(L.status.time) 

103 err = abs(uex - L.uend) 

104 

105 return err, L.status.residual, S.status.iter
