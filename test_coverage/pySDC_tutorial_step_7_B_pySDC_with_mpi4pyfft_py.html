12def run_simulation(spectral=None, ml=None, num_procs=None): 

13 """ 

14 A test program to do SDC, MLSDC and PFASST runs for the 2D NLS equation 

15 

16 Args: 

17 spectral (bool): run in real or spectral space 

18 ml (bool): single or multiple levels 

19 num_procs (int): number of parallel processors 

20 """ 

21 

22 comm = MPI.COMM_WORLD 

23 rank = comm.Get_rank() 

24 

25 # initialize level parameters 

26 level_params = dict() 

27 level_params['restol'] = 1E-08 

28 level_params['dt'] = 1E-01 / 2 

29 level_params['nsweeps'] = [1] 

30 

31 # initialize sweeper parameters 

32 sweeper_params = dict() 

33 sweeper_params['collocation_class'] = CollGaussRadau_Right 

34 sweeper_params['num_nodes'] = [3] 

35 sweeper_params['QI'] = ['LU'] # For the IMEX sweeper, the LU-trick can be activated for the implicit part 

36 sweeper_params['initial_guess'] = 'zero' 

37 

38 # initialize problem parameters 

39 problem_params = dict() 

40 if ml: 

41 problem_params['nvars'] = [(128, 128), (32, 32)] 

42 else: 

43 problem_params['nvars'] = [(128, 128)] 

44 problem_params['spectral'] = spectral 

45 problem_params['comm'] = comm 

46 

47 # initialize step parameters 

48 step_params = dict() 

49 step_params['maxiter'] = 50 

50 

51 # initialize controller parameters 

52 controller_params = dict() 

53 controller_params['logger_level'] = 30 if rank == 0 else 99 

54 # controller_params['predict_type'] = 'fine_only' 

55 

56 # fill description dictionary for easy step instantiation 

57 description = dict() 

58 description['problem_params'] = problem_params # pass problem parameters 

59 description['problem_class'] = nonlinearschroedinger_imex 

60 description['sweeper_class'] = imex_1st_order 

61 description['sweeper_params'] = sweeper_params # pass sweeper parameters 

62 description['level_params'] = level_params # pass level parameters 

63 description['step_params'] = step_params # pass step parameters 

64 description['space_transfer_class'] = fft_to_fft 

65 

66 # set time parameters 

67 t0 = 0.0 

68 Tend = 1.0 

69 

70 f = None 

71 if rank == 0: 

72 f = open('step_7_B_out.txt', 'a') 

73 out = f'Running with ml={ml} and num_procs={num_procs}...' 

74 f.write(out + '\n') 

75 print(out) 

76 

77 # instantiate controller 

78 controller = controller_nonMPI(num_procs=num_procs, controller_params=controller_params, description=description) 

79 

80 # get initial values on finest level 

81 P = controller.MS[0].levels[0].prob 

82 uinit = P.u_exact(t0) 

83 

84 # call main function to get things done... 

85 uend, stats = controller.run(u0=uinit, t0=t0, Tend=Tend) 

86 uex = P.u_exact(Tend) 

87 err = abs(uex - uend) 

88 

89 if rank == 0: 

90 # filter statistics by type (number of iterations) 

91 filtered_stats = filter_stats(stats, type='niter') 

92 

93 # convert filtered statistics to list of iterations count, sorted by process 

94 iter_counts = sort_stats(filtered_stats, sortby='time') 

95 

96 niters = np.array([item[1] for item in iter_counts]) 

97 out = f' Min/Mean/Max number of iterations: ' \ 

98 f'{np.min(niters):4.2f} / {np.mean(niters):4.2f} / {np.max(niters):4.2f}' 

99 f.write(out + '\n') 

100 print(out) 

101 out = ' Range of values for number of iterations: %2i ' % np.ptp(niters) 

102 f.write(out + '\n') 

103 print(out) 

104 out = ' Position of max/min number of iterations: %2i -- %2i' % \ 

105 (int(np.argmax(niters)), int(np.argmin(niters))) 

106 f.write(out + '\n') 

107 print(out) 

108 out = ' Std and var for number of iterations: %4.2f -- %4.2f' % (float(np.std(niters)), float(np.var(niters))) 

109 f.write(out + '\n') 

110 print(out) 

111 

112 out = f'Error: {err:6.4e}' 

113 f.write(out + '\n') 

114 print(out) 

115 

116 timing = sort_stats(filter_stats(stats, type='timing_run'), sortby='time') 

117 out = f'Time to solution: {timing[0][1]:6.4f} sec.' 

118 f.write(out + '\n') 

119 print(out) 

120 

121 assert err <= 1.133E-05, 'Error is too high, got %s' % err 

122 if ml: 

123 if num_procs > 1: 

124 maxmean = 12.5 

125 else: 

126 maxmean = 6.6 

127 else: 

128 maxmean = 12.7 

129 assert np.mean(niters) <= maxmean, 'Mean number of iterations is too high, got %s' % np.mean(niters) 

130 

131 f.write('\n') 

132 print() 

133 f.close()
