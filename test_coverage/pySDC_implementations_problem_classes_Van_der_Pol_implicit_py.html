74 """ 

75 Simple Newton solver for the nonlinear system 

76 

77 Args: 

78 rhs (dtype_f): right-hand side for the nonlinear system 

79 dt (float): abbrev. for the node-to-node stepsize (or any other factor required)

82

84 dtype_u: solution u 

85 """

87 mu = self.params.mu 

88 

89 # create new mesh object from u0 and set initial values for iteration 

90 u = self.dtype_u(u0) 

91 x1 = u[0] 

92 x2 = u[1]

94 # start newton iteration 

95 n = 0 

96 res = 99 

97 while n < self.params.newton_maxiter: 

98 

99 # form the function g with g(u) = 0 

100 g = np.array([x1 - dt * x2 - rhs[0], x2 - dt * (mu * (1 - x1 ** 2) * x2 - x1) - rhs[1]]) 

101 

102 # if g is close to 0, then we are done 

103 res = np.linalg.norm(g, np.inf) 

104 if res < self.params.newton_tol or np.isnan(res): 

105 break 

106 

107 # prefactor for dg/du 

108 c = 1.0 / (-2 * dt ** 2 * mu * x1 * x2 - dt ** 2 - 1 + dt * mu * (1 - x1 ** 2)) 

109 # assemble dg/du 

110 dg = c * np.array([[dt * mu * (1 - x1 ** 2) - 1, -dt], [2 * dt * mu * x1 * x2 + dt, -1]]) 

111 

112 # newton update: u1 = u0 - g/dg 

113 u -= np.dot(dg, g) 

114 

115 # set new values and increase iteration count 

116 x1 = u[0] 

117 x2 = u[1] 

118 n += 1 

119 

120 if np.isnan(res) and self.params.stop_at_nan: 

121 raise ProblemError('Newton got nan after %i iterations, aborting...' % n) 

122 elif np.isnan(res): 

123 self.logger.warning('Newton got nan after %i iterations...' % n) 

124 

125 if n == self.params.newton_maxiter: 

126 raise ProblemError('Newton did not converge after %i iterations, error is %s' % (n, res)) 

127 

128 return u
