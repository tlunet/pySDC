61def run_accuracy_check(prob, coll, dt_list): 

62 """ 

63 Routine to build and solve the linear collocation problem 

64 

65 Args: 

66 prob: a problem instance 

67 coll: a collocation instance 

68 dt_list: list of time-step sizes 

69 

70 Return: 

71 the analytic error of the solved collocation problem 

72 """ 

73 

74 results = {} 

75 # loop over all nvars 

76 for dt in dt_list: 

77 # shrink collocation matrix: first line and column deals with initial value, not needed here 

78 Q = coll.Qmat[1:, 1:] 

79 

80 # build system matrix M of collocation problem 

81 M = sp.eye(prob.params.nvars * coll.num_nodes) - dt * sp.kron(Q, prob.A) 

82 

83 # get initial value at t0 = 0 

84 u0 = prob.u_exact(t=0) 

85 # fill in u0-vector as right-hand side for the collocation problem 

86 u0_coll = np.kron(np.ones(coll.num_nodes), u0.values) 

87 # get exact solution at Tend = dt 

88 uend = prob.u_exact(t=dt) 

89 

90 # solve collocation problem directly 

91 u_coll = sp.linalg.spsolve(M, u0_coll) 

92 

93 # compute error 

94 err = np.linalg.norm(u_coll[-prob.params.nvars:] - uend.values, np.inf) 

95 # get id for this dt and store error in results 

96 id = ID(dt=dt) 

97 results[id] = err 

98 

99 # add list of dt to results for easier access 

100 results['dt_list'] = dt_list 

101 return results

104def get_accuracy_order(results): 

105 """ 

106 Routine to compute the order of accuracy in time 

107 

108 Args: 

109 results: the dictionary containing the errors 

110 

111 Returns: 

112 the list of orders 

113 """ 

114 

115 # retrieve the list of dt from results 

116 assert 'dt_list' in results, 'ERROR: expecting the list of dt in the results dictionary' 

117 dt_list = sorted(results['dt_list'], reverse=True) 

118 

119 order = [] 

120 # loop over two consecutive errors/dt pairs 

121 for i in range(1, len(dt_list)): 

122 # get ids 

123 id = ID(dt=dt_list[i]) 

124 id_prev = ID(dt=dt_list[i - 1]) 

125 

126 # compute order as log(prev_error/this_error)/log(this_dt/old_dt) <-- depends on the sorting of the list! 

127 tmp = np.log(results[id] / results[id_prev]) / np.log(dt_list[i] / dt_list[i - 1]) 

128 order.append(tmp) 

129 

130 return order
