65 beta = 0.0

67 diag = np.zeros(M - 1) 

68 subdiag = np.zeros(M - 2) 

69 

70 diag[0] = (beta - alpha) / (2 + alpha + beta) 

71 

72 for jj in range(1, M - 1): 

73 diag[jj] = (beta - alpha) * (alpha + beta) / (2 * jj + 2 + alpha + beta) / (2 * jj + alpha + beta) 

74 num = np.sqrt(4 * jj * (jj + alpha) * (jj + beta) * (jj + alpha + beta)) 

75 denom = np.sqrt((2 * jj - 1 + alpha + beta) * (2 * jj + alpha + beta) ** 2 * (2 * jj + 1 + alpha + beta)) 

76 subdiag[jj - 1] = num / denom 

77

79 subdiag2 = np.zeros(M - 1) 

80 subdiag1[0:-1] = subdiag

83 Mat = sp.spdiags(data=[subdiag1, diag, subdiag2], diags=[-1, 0, 1], m=M - 1, n=M - 1).todense()
