13def show_residual_across_simulation(stats, fname='residuals.png'): 

14 """ 

15 Helper routine to visualize the residuals across the simulation (one block of PFASST) 

16 

17 Args: 

18 stats (dict): statistics object from a PFASST run 

19 fname (str): filename 

20 """ 

21 

22 # get residuals of the run 

23 extract_stats = filter_stats(stats, type='residual_post_iteration') 

24 

25 # find boundaries for x-,y- and c-axis as well as arrays 

26 maxprocs = 0 

27 maxiter = 0 

28 minres = 0 

29 maxres = -99 

30 for k, v in extract_stats.items(): 

31 maxprocs = max(maxprocs, getattr(k, 'process')) 

32 maxiter = max(maxiter, getattr(k, 'iter')) 

33 minres = min(minres, np.log10(v)) 

34 maxres = max(maxres, np.log10(v)) 

35 

36 # grep residuals and put into array 

37 residual = np.zeros((maxiter, maxprocs + 1)) 

38 residual[:] = -99 

39 for k, v in extract_stats.items(): 

40 step = getattr(k, 'process') 

41 iter = getattr(k, 'iter') 

42 if iter is not -1: 

43 residual[iter - 1, step] = np.log10(v) 

44 

45 # Set up latex stuff and fonts 

46 rc('font', **{"sans-serif": ["Arial"], "size": 30}) 

47 rc('legend', fontsize='small') 

48 rc('xtick', labelsize='small') 

49 rc('ytick', labelsize='small') 

50 

51 # create plot and save 

52 fig, ax = plt.subplots(figsize=(15, 10)) 

53 

54 cmap = plt.get_cmap('Reds') 

55 plt.pcolor(residual.T, cmap=cmap, vmin=minres, vmax=maxres) 

56 

57 cax = plt.colorbar()

59 

60 ax.set_xlabel('iteration') 

61 ax.set_ylabel('process')

63 ax.set_xticks(np.arange(maxiter) + 0.5, minor=False)
