45 (item[0], item[1], abs(base_energy - item[1]) / base_energy) 

46 f.write(out + '\n') 

47 print(out) 

48 for item in energy_mlsdc: 

49 out = 'Total energy and relative deviation in iteration %2i: %12.10f -- %12.8e' % \ 

50 (item[0], item[1], abs(base_energy - item[1]) / base_energy) 

51 f.write(out + '\n') 

52 print(out) 

53 for item in energy_mlsdc_finter: 

54 out = 'Total energy and relative deviation in iteration %2i: %12.10f -- %12.8e' % \ 

55 (item[0], item[1], abs(base_energy - item[1]) / base_energy) 

56 f.write(out + '\n') 

57 print(out) 

58 f.close()

68def run_penning_trap_simulation(mlsdc, finter=False): 

69 """ 

70 A simple test program to run IMEX SDC for a single time step 

71 """ 

72 # initialize level parameters 

73 level_params = dict() 

74 level_params['restol'] = 1E-07 

75 level_params['dt'] = 1.0 / 8 

76 

77 # initialize sweeper parameters 

78 sweeper_params = dict() 

79 sweeper_params['collocation_class'] = CollGaussRadau_Right 

80 sweeper_params['num_nodes'] = 5 

81 

82 # initialize problem parameters for the Penning trap 

83 problem_params = dict() 

84 problem_params['omega_E'] = 4.9 # E-field frequency 

85 problem_params['omega_B'] = 25.0 # B-field frequency 

86 problem_params['u0'] = np.array([[10, 0, 0], [100, 0, 100], [1], [1]]) # initial center of positions 

87 problem_params['nparts'] = 50 # number of particles in the trap 

88 problem_params['sig'] = 0.1 # smoothing parameter for the forces 

89 

90 # initialize step parameters 

91 step_params = dict() 

92 step_params['maxiter'] = 20 

93 

94 # initialize controller parameters 

95 controller_params = dict() 

96 controller_params['hook_class'] = particle_hook # specialized hook class for more statistics and output 

97 controller_params['logger_level'] = 30 

98 

99 transfer_params = dict() 

100 transfer_params['finter'] = finter 

101 

102 # Fill description dictionary for easy hierarchy creation 

103 description = dict() 

104 if mlsdc: 

105 # MLSDC: provide list of two problem classes: one for the fine, one for the coarse level 

106 description['problem_class'] = [penningtrap, penningtrap_coarse] 

107 else: 

108 # SDC: provide only one problem class 

109 description['problem_class'] = penningtrap 

110 description['problem_params'] = problem_params 

111 description['sweeper_class'] = boris_2nd_order 

112 description['sweeper_params'] = sweeper_params 

113 description['level_params'] = level_params 

114 description['step_params'] = step_params 

115 description['space_transfer_class'] = particles_to_particles 

116 description['base_transfer_params'] = transfer_params 

117 

118 # instantiate the controller (no controller parameters used here) 

119 controller = controller_nonMPI(num_procs=1, controller_params=controller_params, description=description) 

120 

121 # set time parameters 

122 t0 = 0.0 

123 Tend = level_params['dt'] 

124 

125 # get initial values on finest level 

126 P = controller.MS[0].levels[0].prob 

127 uinit = P.u_init() 

128 

129 # call and time main function to get things done... 

130 start_time = time.time() 

131 uend, stats = controller.run(u0=uinit, t0=t0, Tend=Tend) 

132 end_time = time.time() - start_time 

133 

134 return stats, end_time

138 main()
