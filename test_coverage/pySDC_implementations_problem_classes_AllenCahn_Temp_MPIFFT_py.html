11class allencahn_temp_imex(ptype): 

12 """ 

13 Example implementing Allen-Cahn equation in 2-3D using mpi4py-fft for solving linear parts, IMEX time-stepping 

14 

15 mpi4py-fft: https://mpi4py-fft.readthedocs.io/en/latest/ 

16 

17 Attributes: 

18 fft: fft object 

19 X: grid coordinates in real space 

20 K2: Laplace operator in spectral space 

21 dx: mesh width in x direction 

22 dy: mesh width in y direction 

23 """

25 def __init__(self, problem_params, dtype_u=mesh, dtype_f=imex_mesh): 

26 """ 

27 Initialization routine 

28 

29 Args: 

30 problem_params (dict): custom parameters for the example 

31 dtype_u: fft data type (will be passed to parent class) 

32 dtype_f: fft data type wuth implicit and explicit parts (will be passed to parent class) 

33 """ 

34 

35 if 'L' not in problem_params: 

36 problem_params['L'] = 1.0 

37 if 'init_type' not in problem_params: 

38 problem_params['init_type'] = 'circle' 

39 if 'comm' not in problem_params: 

40 problem_params['comm'] = None 

41 if 'dw' not in problem_params: 

42 problem_params['dw'] = 1.0 

43 

44 # these parameters will be used later, so assert their existence 

45 essential_keys = ['nvars', 'eps', 'L', 'radius', 'dw', 'spectral', 'TM', 'D'] 

46 for key in essential_keys: 

47 if key not in problem_params: 

48 msg = 'need %s to instantiate problem, only got %s' % (key, str(problem_params.keys())) 

49 raise ParameterError(msg) 

50 

51 if not (isinstance(problem_params['nvars'], tuple) and len(problem_params['nvars']) > 1): 

52 raise ProblemError('Need at least two dimensions') 

53 

54 # creating FFT structure 

55 ndim = len(problem_params['nvars']) 

56 axes = tuple(range(ndim)) 

57 self.fft = PFFT(problem_params['comm'], list(problem_params['nvars']), axes=axes, dtype=np.float, collapse=True) 

58 

59 # get test data to figure out type and dimensions 

60 tmp_u = newDistArray(self.fft, problem_params['spectral']) 

61 

62 # add two components to contain field and temperature 

63 self.ncomp = 2 

64 sizes = tmp_u.shape + (self.ncomp,) 

65 

66 # invoke super init, passing the communicator and the local dimensions as init 

67 super(allencahn_temp_imex, self).__init__(init=(sizes, problem_params['comm'], tmp_u.dtype), 

68 dtype_u=dtype_u, dtype_f=dtype_f, params=problem_params) 

69 

70 L = np.array([self.params.L] * ndim, dtype=float) 

71 

72 # get local mesh 

73 X = np.ogrid[self.fft.local_slice(False)] 

74 N = self.fft.global_shape() 

75 for i in range(len(N)): 

76 X[i] = (X[i] * L[i] / N[i]) 

77 self.X = [np.broadcast_to(x, self.fft.shape(False)) for x in X] 

78 

79 # get local wavenumbers and Laplace operator 

80 s = self.fft.local_slice() 

81 N = self.fft.global_shape() 

82 k = [np.fft.fftfreq(n, 1. / n).astype(int) for n in N[:-1]] 

83 k.append(np.fft.rfftfreq(N[-1], 1. / N[-1]).astype(int)) 

84 K = [ki[si] for ki, si in zip(k, s)] 

85 Ks = np.meshgrid(*K, indexing='ij', sparse=True) 

86 Lp = 2 * np.pi / L 

87 for i in range(ndim): 

88 Ks[i] = (Ks[i] * Lp[i]).astype(float) 

89 K = [np.broadcast_to(k, self.fft.shape(True)) for k in Ks] 

90 K = np.array(K).astype(float) 

91 self.K2 = np.sum(K * K, 0, dtype=float) 

92 

93 # Need this for diagnostics 

94 self.dx = self.params.L / problem_params['nvars'][0] 

95 self.dy = self.params.L / problem_params['nvars'][1]

97 def eval_f(self, u, t): 

98 """ 

99 Routine to evaluate the RHS 

100 

101 Args: 

102 u (dtype_u): current values 

103 t (float): current time 

104 

105 Returns: 

106 dtype_f: the RHS 

107 """ 

108 

109 f = self.dtype_f(self.init) 

110 

111 if self.params.spectral: 

112 

113 f.impl[..., 0] = -self.K2 * u[..., 0] 

114 

115 if self.params.eps > 0: 

116 tmp_u = newDistArray(self.fft, False) 

117 tmp_T = newDistArray(self.fft, False) 

118 tmp_u = self.fft.backward(u[..., 0], tmp_u) 

119 tmp_T = self.fft.backward(u[..., 1], tmp_T) 

120 tmpf = - 2.0 / self.params.eps ** 2 * tmp_u * (1.0 - tmp_u) * (1.0 - 2.0 * tmp_u) - \ 

121 6.0 * self.params.dw * (tmp_T - self.params.TM) / self.params.TM * tmp_u * (1.0 - tmp_u) 

122 f.expl[..., 0] = self.fft.forward(tmpf) 

123 

124 f.impl[..., 1] = -self.params.D * self.K2 * u[..., 1] 

125 f.expl[..., 1] = -f.impl[..., 0] - f.expl[..., 0] 

126 

127 else: 

128 

129 u_hat = self.fft.forward(u[..., 0]) 

130 lap_u_hat = -self.K2 * u_hat 

131 f.impl[..., 0] = self.fft.backward(lap_u_hat, f.impl[..., 0]) 

132 

133 if self.params.eps > 0: 

134 f.expl[..., 0] = -2.0 / self.params.eps ** 2 * u[..., 0] * (1.0 - u[..., 0]) * (1.0 - 2.0 * u[..., 0]) 

135 f.expl[..., 0] -= 6.0 * self.params.dw * (u[..., 1] - self.params.TM) / self.params.TM * u[..., 0] * \ 

136 (1.0 - u[..., 0]) 

137 

138 u_hat = self.fft.forward(u[..., 1]) 

139 lap_u_hat = -self.params.D * self.K2 * u_hat 

140 f.impl[..., 1] = self.fft.backward(lap_u_hat, f.impl[..., 1]) 

141 f.expl[..., 1] = -f.impl[..., 0] - f.expl[..., 0] 

142 

143 return f

145 def solve_system(self, rhs, factor, u0, t): 

146 """ 

147 Simple FFT solver for the diffusion part 

148 

149 Args: 

150 rhs (dtype_f): right-hand side for the linear system 

151 factor (float) : abbrev. for the node-to-node stepsize (or any other factor required) 

152 u0 (dtype_u): initial guess for the iterative solver (not used here so far) 

153 t (float): current time (e.g. for time-dependent BCs) 

154 

155 Returns: 

156 dtype_u: solution as mesh 

157 """ 

158 

159 if self.params.spectral: 

160 

161 me = self.dtype_u(self.init) 

162 me[..., 0] = rhs[..., 0] / (1.0 + factor * self.K2) 

163 me[..., 1] = rhs[..., 1] / (1.0 + factor * self.params.D * self.K2) 

164 

165 else: 

166 

167 me = self.dtype_u(self.init) 

168 rhs_hat = self.fft.forward(rhs[..., 0]) 

169 rhs_hat /= (1.0 + factor * self.K2) 

170 me[..., 0] = self.fft.backward(rhs_hat) 

171 rhs_hat = self.fft.forward(rhs[..., 1]) 

172 rhs_hat /= (1.0 + factor * self.params.D * self.K2) 

173 me[..., 1] = self.fft.backward(rhs_hat) 

174 

175 return me

177 def u_exact(self, t): 

178 """ 

179 Routine to compute the exact solution at time t 

180 

181 Args: 

182 t (float): current time 

183 

184 Returns: 

185 dtype_u: exact solution 

186 """

188 def circle(): 

189 tmp_me = newDistArray(self.fft, self.params.spectral) 

190 r2 = (self.X[0] - 0.5) ** 2 + (self.X[1] - 0.5) ** 2 

191 if self.params.spectral: 

192 tmp = 0.5 * (1.0 + np.tanh((self.params.radius - np.sqrt(r2)) / (np.sqrt(2) * self.params.eps))) 

193 tmp_me[:] = self.fft.forward(tmp) 

194 else: 

195 tmp_me[:] = 0.5 * (1.0 + np.tanh((self.params.radius - np.sqrt(r2)) / (np.sqrt(2) * self.params.eps))) 

196 return tmp_me 

197 

198 def circle_rand(): 

199 tmp_me = newDistArray(self.fft, self.params.spectral) 

200 ndim = len(tmp_me.shape) 

201 L = int(self.params.L) 

202 # get random radii for circles/spheres 

203 np.random.seed(1) 

204 lbound = 3.0 * self.params.eps 

205 ubound = 0.5 - self.params.eps 

206 rand_radii = (ubound - lbound) * np.random.random_sample(size=tuple([L] * ndim)) + lbound 

207 # distribute circles/spheres 

208 tmp = newDistArray(self.fft, False) 

209 if ndim == 2: 

210 for i in range(0, L): 

211 for j in range(0, L): 

212 # build radius 

213 r2 = (self.X[0] + i - L + 0.5) ** 2 + (self.X[1] + j - L + 0.5) ** 2 

214 # add this blob, shifted by 1 to avoid issues with adding up negative contributions 

215 tmp += np.tanh((rand_radii[i, j] - np.sqrt(r2)) / (np.sqrt(2) * self.params.eps)) + 1 

216 # normalize to [0,1] 

217 tmp *= 0.5 

218 assert np.all(tmp <= 1.0) 

219 if self.params.spectral: 

220 tmp_me[:] = self.fft.forward(tmp) 

221 else: 

222 tmp_me[:] = tmp[:] 

223 return tmp_me 

224 

225 def sine(): 

226 tmp_me = newDistArray(self.fft, self.params.spectral) 

227 if self.params.spectral: 

228 tmp = np.sin(2 * np.pi * self.X[0]) * np.sin(2 * np.pi * self.X[1]) 

229 tmp_me[:] = self.fft.forward(tmp) 

230 else: 

231 tmp_me[:] = np.sin(2 * np.pi * self.X[0]) * np.sin(2 * np.pi * self.X[1]) 

232 return tmp_me 

233 

234 assert t == 0, 'ERROR: u_exact only valid for t=0' 

235 me = self.dtype_u(self.init, val=0.0) 

236 if self.params.init_type == 'circle': 

237 me[..., 0] = circle() 

238 me[..., 1] = sine() 

239 elif self.params.init_type == 'circle_rand': 

240 me[..., 0] = circle_rand() 

241 me[..., 1] = sine() 

242 else: 

243 raise NotImplementedError('type of initial value not implemented, got %s' % self.params.init_type) 

244 

245 return me
