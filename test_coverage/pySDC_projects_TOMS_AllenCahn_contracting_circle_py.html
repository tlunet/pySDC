81def run_SDC_variant(variant=None, inexact=False): 

82 """ 

83 Routine to run particular SDC variant 

84 

85 Args: 

86 variant (str): string describing the variant 

87 inexact (bool): flag to use inexact nonlinear solve (or nor) 

88 

89 Returns: 

90 timing (float) 

91 niter (float) 

92 """ 

93 

94 # load (incomplete) default parameters 

95 description, controller_params = setup_parameters() 

96 

97 # add stuff based on variant 

98 if variant == 'fully-implicit': 

99 description['problem_class'] = allencahn_fullyimplicit 

100 description['sweeper_class'] = generic_implicit 

101 if inexact: 

102 description['problem_params']['newton_maxiter'] = 1 

103 elif variant == 'semi-implicit': 

104 description['problem_class'] = allencahn_semiimplicit 

105 description['sweeper_class'] = imex_1st_order 

106 if inexact: 

107 description['problem_params']['lin_maxiter'] = 10 

108 elif variant == 'semi-implicit_v2': 

109 description['problem_class'] = allencahn_semiimplicit_v2 

110 description['sweeper_class'] = imex_1st_order 

111 if inexact: 

112 description['problem_params']['newton_maxiter'] = 1 

113 elif variant == 'multi-implicit': 

114 description['problem_class'] = allencahn_multiimplicit 

115 description['sweeper_class'] = multi_implicit 

116 if inexact: 

117 description['problem_params']['newton_maxiter'] = 1 

118 description['problem_params']['lin_maxiter'] = 10 

119 elif variant == 'multi-implicit_v2': 

120 description['problem_class'] = allencahn_multiimplicit_v2 

121 description['sweeper_class'] = multi_implicit 

122 if inexact: 

123 description['problem_params']['newton_maxiter'] = 1 

124 else: 

125 raise NotImplemented('Wrong variant specified, got %s' % variant) 

126 

127 if inexact: 

128 out = 'Working on inexact %s variant...' % variant 

129 else: 

130 out = 'Working on exact %s variant...' % variant 

131 print(out) 

132 

133 # setup parameters "in time" 

134 t0 = 0 

135 Tend = 0.032 

136 

137 # instantiate controller 

138 controller = controller_nonMPI(num_procs=1, controller_params=controller_params, description=description) 

139 

140 # get initial values on finest level 

141 P = controller.MS[0].levels[0].prob 

142 uinit = P.u_exact(t0) 

143 

144 # call main function to get things done... 

145 uend, stats = controller.run(u0=uinit, t0=t0, Tend=Tend) 

146 

147 # filter statistics by variant (number of iterations) 

148 filtered_stats = filter_stats(stats, type='niter') 

149 

150 # convert filtered statistics to list of iterations count, sorted by process 

151 iter_counts = sort_stats(filtered_stats, sortby='time') 

152 

153 # compute and print statistics 

154 niters = np.array([item[1] for item in iter_counts]) 

155 out = ' Mean number of iterations: %4.2f' % np.mean(niters) 

156 print(out) 

157 out = ' Range of values for number of iterations: %2i ' % np.ptp(niters) 

158 print(out) 

159 out = ' Position of max/min number of iterations: %2i -- %2i' % \ 

160 (int(np.argmax(niters)), int(np.argmin(niters))) 

161 print(out) 

162 out = ' Std and var for number of iterations: %4.2f -- %4.2f' % (float(np.std(niters)), float(np.var(niters))) 

163 print(out) 

164 

165 print(' Iteration count (nonlinear/linear): %i / %i' % (P.newton_itercount, P.lin_itercount)) 

166 print(' Mean Iteration count per call: %4.2f / %4.2f' % (P.newton_itercount / max(P.newton_ncalls, 1), 

167 P.lin_itercount / max(P.lin_ncalls, 1))) 

168 

169 timing = sort_stats(filter_stats(stats, type='timing_run'), sortby='time') 

170 

171 print('Time to solution: %6.4f sec.' % timing[0][1]) 

172 print() 

173 

174 return stats
